inline auto original::pow(const double base, const int exp) -> double
{
if (base == 0 && exp <= 0)
{
throw valueError();
}
double res = 1.0;
for (int i = 0; i < abs(exp); i += 1)
{
res *= base;
}
return exp >= 0 ? res : 1 / res;
}
