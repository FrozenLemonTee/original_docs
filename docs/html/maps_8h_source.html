593template<typename K_TYPE, typename V_TYPE, typename HASH, typename ALLOC>
594typename original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::Iterator*
595original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::begins() const {
596 auto p_buckets = const_cast<vector<hashNode*, rebind_alloc_pointer>*>(&this->buckets);
597 if (this->buckets[0]) {
598 return new Iterator(p_buckets, 0, this->buckets[0]);
599 }
600 auto bucket = Iterator::findNextValidBucket(p_buckets, 0);
601 return new Iterator(p_buckets, bucket, p_buckets->get(bucket));
602}

604template<typename K_TYPE, typename V_TYPE, typename HASH, typename ALLOC>
605typename original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::Iterator*
606original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::ends() const {
607 auto p_buckets = const_cast<vector<hashNode*, rebind_alloc_pointer>*>(&this->buckets);
608 auto bucket = Iterator::findPrevValidBucket(p_buckets, this->buckets.size());
609 auto node = this->buckets[bucket];
610 while (node && node->getPNext()) {
611 node = node->getPNext();
612 }
613 return new Iterator(p_buckets, bucket, node);
614}
