698typename original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::Iterator*
699original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::begins() const {
700 auto p_buckets = const_cast<vector<hashNode*, rebind_alloc_pointer>*>(&this->buckets);
701 if (this->buckets[0]) {
702 return new Iterator(p_buckets, 0, this->buckets[0]);

705 return new Iterator(p_buckets, bucket, p_buckets->get(bucket));

709typename original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::Iterator*
710original::hashMap<K_TYPE, V_TYPE, HASH, ALLOC>::ends() const {
711 auto p_buckets = const_cast<vector<hashNode*, rebind_alloc_pointer>*>(&this->buckets);
712 auto bucket = Iterator::findPrevValidBucket(p_buckets, this->buckets.size());
713 auto node = this->buckets[bucket];
714 while (node && node->getPNext()) {
715 node = node->getPNext();
716 }
717 return new Iterator(p_buckets, bucket, node);
