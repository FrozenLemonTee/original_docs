360template <typename TYPE>
361void original::objPoolAllocator<TYPE>::chunkAllocate(const u_integer num_element, u_integer index)
362{
363 const u_integer block_size = (1 << index) * CHUNK_SIZE;
364 char* new_free_chunk = nullptr;
365 allocatedChunks* new_allocated_chunk = nullptr;
366
367 try
368 {
369 new_free_chunk = static_cast<char*>(operator new(num_element * block_size));
370 new_allocated_chunk = static_cast<allocatedChunks*>(operator new(sizeof(allocatedChunks)));
371 }
372 catch (const std::bad_alloc&)
373 {
374 throw allocateError();
375 }
376
377 new_allocated_chunk->chunks = new_free_chunk;
378 new_allocated_chunk->next = this->allocated_list_head;
379 this->allocated_list_head = new_allocated_chunk;
380
381 for (u_integer i = 0; i < num_element; i++)
382 {
383 auto cur_ptr = reinterpret_cast<freeChunk*>(new_free_chunk + i * block_size);
384 cur_ptr->next = this->free_list_head[index];
385 this->free_list_head[index] = cur_ptr;
386 }
387 this->chunks_available[index] += num_element;
388 this->chunk_count[index] = this->chunk_count[index] + (this->chunk_count[index] >> 1);
389}

392void original::objPoolAllocator<TYPE>::reset() noexcept {
393 while (this->allocated_list_head) {
394 auto next_chunk = this->allocated_list_head->next;
395 allocators::free(this->allocated_list_head->chunks);
396 allocators::free(this->allocated_list_head);
397 this->allocated_list_head = next_chunk;
398 }
399
400 if (this->chunk_count) {

451template <typename TYPE>
452TYPE* original::objPoolAllocator<TYPE>::allocate(const u_integer size)
453{
454 if (size == 0) {
455 return nullptr;
456 }
457
458 u_integer index = getChunkIndex(size);
459
460 if (index >= this->size_index_max) {
461 return allocators::malloc<TYPE>(size);
462 }
463
464 if (!this->free_list_head[index] || this->chunks_available[index] * (static_cast<u_integer>(1) << index) < size) {
465 this->chunkAllocate(max<u_integer>(1, this->chunk_count[index]), index);
466 }
467
468 auto cur_ptr = this->free_list_head[index];
469 this->free_list_head[index] = this->free_list_head[index]->next;
470 this->chunks_available[index] -= 1;
471 return reinterpret_cast<TYPE*>(cur_ptr);
472}
