343template <typename TYPE>
344void original::objPoolAllocator<TYPE>::chunkAllocate(const u_integer num_element, u_integer index)
345{
346 const u_integer block_size = (1 << index) * CHUNK_SIZE;
347 char* new_free_chunk = nullptr;
348 allocatedChunks* new_allocated_chunk = nullptr;
349
350 try
351 {
352 new_free_chunk = static_cast<char*>(operator new(num_element * block_size));
353 new_allocated_chunk = static_cast<allocatedChunks*>(operator new(sizeof(allocatedChunks)));
354 }
355 catch (const std::bad_alloc&)
356 {
357 throw allocateError();
358 }
359
360 new_allocated_chunk->chunks = new_free_chunk;
361 new_allocated_chunk->next = this->allocated_list_head;
362 this->allocated_list_head = new_allocated_chunk;
363
364 for (u_integer i = 0; i < num_element; i++)
365 {
366 auto cur_ptr = reinterpret_cast<freeChunk*>(new_free_chunk + i * block_size);
367 cur_ptr->next = this->free_list_head[index];
368 this->free_list_head[index] = cur_ptr;
369 }
370 this->chunks_available[index] += num_element;
371 this->chunk_count[index] = this->chunk_count[index] + (this->chunk_count[index] >> 1);
372}

375void original::objPoolAllocator<TYPE>::reset() noexcept {
376 while (this->allocated_list_head) {
377 auto next_chunk = this->allocated_list_head->next;
378 allocators::free(this->allocated_list_head->chunks);
379 allocators::free(this->allocated_list_head);
380 this->allocated_list_head = next_chunk;
381 }
382
383 if (this->chunk_count) {

434template <typename TYPE>
435TYPE* original::objPoolAllocator<TYPE>::allocate(const u_integer size)
436{
437 if (size == 0) {
438 return nullptr;
439 }
440
441 u_integer index = getChunkIndex(size);
442
443 if (index >= this->size_index_max) {
444 return allocators::malloc<TYPE>(size);
445 }
446
447 if (!this->free_list_head[index] || this->chunks_available[index] * (static_cast<u_integer>(1) << index) < size) {
448 this->chunkAllocate(max<u_integer>(1, this->chunk_count[index]), index);
449 }
450
451 auto cur_ptr = this->free_list_head[index];
452 this->free_list_head[index] = this->free_list_head[index]->next;
453 this->chunks_available[index] -= 1;
454 return reinterpret_cast<TYPE*>(cur_ptr);
455}
