313template <typename TYPE>
314void original::objPoolAllocator<TYPE>::chunkAllocate(const u_integer num_element, u_integer index)
315{
316 const u_integer block_size = (1 << index) * CHUNK_SIZE;
317 char* new_free_chunk = nullptr;
318 allocatedChunks* new_allocated_chunk = nullptr;
319
320 try
321 {
322 new_free_chunk = static_cast<char*>(operator new(num_element * block_size));
323 new_allocated_chunk = static_cast<allocatedChunks*>(operator new(sizeof(allocatedChunks)));
324 }
325 catch (const std::bad_alloc&)
326 {
327 throw allocateError();
328 }
329
330 new_allocated_chunk->chunks = new_free_chunk;
331 new_allocated_chunk->next = this->allocated_list_head;
332 this->allocated_list_head = new_allocated_chunk;
333
334 for (u_integer i = 0; i < num_element; i++)
335 {
336 auto cur_ptr = reinterpret_cast<freeChunk*>(new_free_chunk + i * block_size);
337 cur_ptr->next = this->free_list_head[index];
338 this->free_list_head[index] = cur_ptr;
339 }
340 this->chunks_available[index] += num_element;
341 this->chunk_count[index] = this->chunk_count[index] + (this->chunk_count[index] >> 1);
342}
