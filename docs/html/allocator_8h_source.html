398template <typename TYPE>
399void original::objPoolAllocator<TYPE>::chunkAllocate(const u_integer num_element, u_integer index)
400{
401 const u_integer block_size = (1 << index) * CHUNK_SIZE;
402 auto new_free_chunk = allocators::malloc<char>(num_element * block_size);
403 auto new_allocated_chunk = allocators::malloc<allocatedChunks>(1);
404
405 new_allocated_chunk->chunks = new_free_chunk;
406 new_allocated_chunk->next = this->allocated_list_head;
407 this->allocated_list_head = new_allocated_chunk;
408
409 for (u_integer i = 0; i < num_element; i++)
410 {
411 auto cur_ptr = reinterpret_cast<freeChunk*>(new_free_chunk + i * block_size);
412 cur_ptr->next = this->free_list_head[index];
413 this->free_list_head[index] = cur_ptr;
414 }
415 this->chunks_available[index] += num_element;
416 this->chunk_count[index] = this->chunk_count[index] + (this->chunk_count[index] >> 1);
417}

472template <typename TYPE>
473TYPE* original::objPoolAllocator<TYPE>::allocate(const u_integer size)
474{
475 if (size == 0) {
476 return nullptr;
477 }
478
479 u_integer index = getChunkIndex(size);
480
481 if (index >= this->size_class_count) {
482 return allocators::malloc<TYPE>(size);
483 }
484
485 if (!this->free_list_head[index] || this->chunks_available[index] * (static_cast<u_integer>(1) << index) < size) {
486 this->chunkAllocate(max<u_integer>(1, this->chunk_count[index]), index);
487 }
488
489 auto cur_ptr = this->free_list_head[index];
490 this->free_list_head[index] = this->free_list_head[index]->next;
491 this->chunks_available[index] -= 1;
492 return reinterpret_cast<TYPE*>(cur_ptr);
493}
