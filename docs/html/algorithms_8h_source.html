1028 auto original::algorithms::heapAdjustDown(const iterator<TYPE>& begin, const iterator<TYPE>& range,
1029 const iterator<TYPE>& current, const Callback& compares) -> void
1030 {

1032 return;
1033

1041 break;

1044 delete it;

1046 delete child;
1047 }
