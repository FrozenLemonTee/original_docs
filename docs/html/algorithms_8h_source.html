767 auto original::algorithms::heapAdjustDown(const iterator<TYPE>& begin, const iterator<TYPE>& range,
768 const iterator<TYPE>& current, const Callback& compares) -> void
769 {
770 if (distance(current, begin) < 0)
771 return;
772
773 auto* it = current.clone();
774 while (distance(*it, begin) * 2 + 1 <= distance(range, begin))
775 {
776 auto* child = heapGetPrior(begin, range, *it, compares);
777 if (compare(it, child, compares))
778 {
779 delete child;
780 break;

782 swap(it, child);
783 delete it;

785 delete child;
786 }
787 delete it;
