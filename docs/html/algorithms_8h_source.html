880 template <typename TYPE, typename Callback>
881 requires original::Compare<Callback, TYPE>
882 auto original::algorithms::heapAdjustDown(const iterator<TYPE>& begin, const iterator<TYPE>& range,
883 const iterator<TYPE>& current, const Callback& compares) -> void
884 {
885 if (distance(current, begin) < 0)
886 return;
887
888 auto* it = current.clone();
889 while (distance(*it, begin) * 2 + 1 <= distance(range, begin))
890 {
891 auto* child = heapGetPrior(begin, range, *it, compares);
892 if (compare(it, child, compares))
893 {
894 delete child;
895 break;
896 }
897 swap(it, child);
898 delete it;
899 it = child->clone();
900 delete child;
901 }
902 delete it;
903 }
