1162 template <typename TYPE, typename Callback>

1164 auto original::algorithms::heapAdjustDown(const iterator<TYPE>& begin, const iterator<TYPE>& range,
1165 const iterator<TYPE>& current, const Callback& compares) -> void
1166 {
1167 if (distance(current, begin) < 0)
1168 return;
1169
1170 auto* it = current.clone();

1172 {
