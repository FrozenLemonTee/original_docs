304 template <typename TYPE>
305 auto original::filterStream<TYPE>::toPostFix() const -> void{
306 this->flag = true;
307
308 chain<std::shared_ptr<filter<TYPE>>> stream_post;
309 chain<opts> ops_post;
310 chain<opts> ops_tmp;
311
312 auto it_stream = this->stream.begins();
313 auto it_ops = this->ops.begins();
314
315 while (it_stream->isValid()){
316 if (it_stream->get() != nullptr){
317 stream_post.pushEnd(it_stream->get());
318 } else if (it_ops->isValid()){
319 switch (it_ops->get()) {
320 case opts::LEFT_BRACKET:
321 ops_tmp.pushEnd(opts::LEFT_BRACKET);
322 break;
323 case opts::RIGHT_BRACKET:
324 while (!ops_tmp.empty() && ops_tmp[-1] != opts::LEFT_BRACKET){
325 stream_post.pushEnd(nullptr);
326 ops_post.pushEnd(ops_tmp.popEnd());
327 }
328 ops_tmp.popEnd();
329 break;
330 case opts::NOT:
331 ops_tmp.pushEnd(opts::NOT);
332 break;
333 default:
334 while (!ops_tmp.empty()
335 && ops_tmp[-1] >= it_ops->get()
336 && ops_tmp[-1] != opts::LEFT_BRACKET){
337 stream_post.pushEnd(nullptr);
338 ops_post.pushEnd(ops_tmp.popEnd());
339 }
340 ops_tmp.pushEnd(it_ops->get());
341 break;
342 }
343 it_ops->next();
344 }
345 it_stream->next();
346 }
347
348 while (!ops_tmp.empty()){
349 stream_post.pushEnd(nullptr);
350 ops_post.pushEnd(ops_tmp.popEnd());
351 }
352 this->stream = stream_post;
353 this->ops = ops_post;
354 delete it_ops;
355 delete it_stream;
356 }

475 template <typename TYPE>
476 auto original::filterStream<TYPE>::operator()(const TYPE &t) const -> bool {
477 if (!this->flag) this->toPostFix();
478
479 chain<bool> value_stack;
480 auto it_stream = this->stream.begins();
481 auto it_ops = this->ops.begins();
482
483 while (it_stream->isValid()){
484 if (it_stream->get() != nullptr){
485 value_stack.pushEnd(it_stream->get()->operator()(t));
486 } else if (it_ops->isValid()){
487 switch (it_ops->get()) {
488 case opts::NOT:
489 value_stack[-1] = !value_stack[-1];
490 break;
491 default:
492 const bool right = value_stack.popEnd();
493 const bool left = value_stack.popEnd();
494 value_stack.pushEnd(it_ops->get() == opts::AND ?
495 left && right : left || right);
496 break;
497 }
498 it_ops->next();
499 }
500 it_stream->next();
501 }
502 delete it_ops;
503 delete it_stream;
504 return value_stack[-1];
505 }
