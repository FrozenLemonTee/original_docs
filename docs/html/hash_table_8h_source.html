698original::hashTable<K_TYPE, V_TYPE, ALLOC, HASH>::bucketsCopy(const buckets_type& buckets) const
699{
700 buckets_type new_buckets = buckets_type(buckets.size(), rebind_alloc_pointer{}, nullptr);
701 for (u_integer i = 0; i < buckets.size(); ++i) {
702 hashNode* old_node = buckets[i];
703 hashNode* prev_new_node = nullptr;
704
705 while (old_node) {
706 hashNode* new_node = this->createNode(old_node->getKey(), old_node->getValue());

708 if (!prev_new_node) {
709 new_buckets[i] = new_node;

711 prev_new_node->setPNext(new_node);

713
714 prev_new_node = new_node;
715 old_node = old_node->getPNext();
716 }
717 }
718 return new_buckets;

781template<typename K_TYPE, typename V_TYPE, typename ALLOC, typename HASH>
782void original::hashTable<K_TYPE, V_TYPE, ALLOC, HASH>::rehash(u_integer new_bucket_count) {
783 if (new_bucket_count == this->getBucketCount())
784 return;
785
786 auto new_buckets = buckets_type(new_bucket_count, rebind_alloc_pointer{}, nullptr);
787
788 for (hashNode*& old_head : this->buckets) {
789 while (old_head) {
790 hashNode* cur = old_head;
791 old_head = old_head->getPNext();
792
793 cur->setPNext(nullptr);
794 auto code = this->hash_(cur->getKey()) % new_bucket_count;
795
796 cur->setPNext(new_buckets[code]);
797 new_buckets[code] = cur;
798 }
799 }
800
801 this->buckets = std::move(new_buckets);
802}

864bool original::hashTable<K_TYPE, V_TYPE, ALLOC, HASH>::erase(const K_TYPE &key) {
865 if (this->size_ == 0)
866 return false;
867
868 this->adjust();
869
870 u_integer code = this->getHashCode(key);
871 hashNode* cur = this->buckets[code];
872 hashNode* prev = nullptr;
873
874 while (cur){
875 if (cur->getKey() == key) {
876 if (prev) {
877 hashNode::connect(prev, cur->getPNext());

883 return true;

886 cur = cur->getPNext();
887 }
