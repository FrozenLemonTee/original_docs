190 const time::point start = time::point::now();
191 while (!predicate()) {
192 auto elapsed = time::point::now() - start;
193 if (elapsed >= d)
194 return false;
195 if (!this->waitFor(mutex, d - elapsed))
196 return false;
197 if (predicate())
198 return true;
