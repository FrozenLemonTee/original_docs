240bool original::conditionBase::waitFor(mutexBase& mutex, const time::duration& d, Pred predicate) noexcept(noexcept(predicate())) {
241 const time::point start = time::point::now();
242 while (!predicate()) {
243 auto elapsed = time::point::now() - start;
244 if (elapsed >= d)
245 return false;
246 if (!this->waitFor(mutex, d - elapsed))
247 return false;
248 if (predicate())
249 return true;
250 }
251 return true;
